INT temp;
LINT impar[3];
INT i;
INT num;

ENQUANTO(i < 3) {
    ESCREVER("Introduza um valor: ");
    LER(temp);

    SE (temp%2!=0){
        impar[num] <- temp;
        num <- num + 1;
    }
    
    i <- i+1;
}

i <- 0;

ESCREVER("Impares: ( ");
ENQUANTO(i < num) {
    ESCREVER(impar[i]);
    ESCREVER(", ");
    i <- i+1;
}

ESCREVER(" )\n");
ESCREVER("Introduziu ", num);
